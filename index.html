async function updateImperialData() { try { const response = await fetch('revenue_data.json?t=' + new Date().getTime()); if (!response.ok) throw new Error('Wait...'); const data = await response.json(); document.getElementById('mrcDisplay').innerText = data.total_mrc.toFixed(4); document.getElementById('lastUpdate').innerText = "최근 채굴: " + data.last_update; console.log("Imperial Data Sync!", data.total_mrc); } catch (e) { console.log("Loading..."); } } setInterval(updateImperialData, 3000); updateImperialData();
